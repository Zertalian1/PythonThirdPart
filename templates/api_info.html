Endpoints:

University:

POST: api/university/create - create new university

GET: api/university/create - get create university page

GET: api/university - get all universities

GET: api/university/{id} - get university by id

GET: api/university/delete/{id} - delete university by id

GET: api/university/edit/{id} - get edit university page

POST: api/university/edit/{id} - edit university by id

Student:

POST: api/create-student - create new student

GET: api/create-student - get create student page

GET: api/student - get all students

GET: api/student/{id} - get student by id

DELETE: api/student/{id} - delete student by id

PATCH: api/student/{id} - edit student by id
